// 1초마다 상태 저장 setInterval(() => { localStorage.setItem('pomodoro-state', JSON.stringify({ mode: currentMode, timeRemaining: timeRemaining, roundCount: roundCount, isRunning: isRunning, focusMinutes: focusMinutes, breakMinutes: breakMinutes, lastTimestamp: Date.now() // 마지막 저장 시각 })); }, 1000); // 페이지 로드 시 복구 function loadState() { const saved = localStorage.getItem('pomodoro-state'); if (!saved) return; const state = JSON.parse(saved); // 작동 중이었으면 = 떠난 시간만큼 차감 if (state.isRunning) { const elapsed = Math.floor((Date.now() - state.lastTimestamp) / 1000); state.timeRemaining -= elapsed; // 떠난 동안 시간 다 끝났으면 자동 전환 while (state.timeRemaining <= 0) { if (state.mode === 'focus') { state.mode = 'break'; state.timeRemaining += state.breakMinutes * 60; } else { state.mode = 'focus'; state.timeRemaining += state.focusMinutes * 60; state.roundCount++; } } } // 상태 적용 currentMode = state.mode; timeRemaining = state.timeRemaining; roundCount = state.roundCount; // ... }
